142  % minimize 1/2 * (q-qbar)^2
143  % subject to Phi( q, tbar ) = 0

145  function projected_pos = project_position( self, t, pos )
146  [n,m] = self.DAEclass.getDim();
147  q = pos;
148  mu = zeros(m,1);
149  tol = norm(pos,inf)*sqrt(eps);
150  for k=1:10
151  phi = self.DAEclass.Phi( t, q );
152  M = self.DAEclass.Phi_q( t, q );
153  Dql = [ eye(n), M.'; M, zeros(m,m) ] \ [ (pos-q) - (M.' * mu);-phi];
154  q = q + Dql(1:n);
155  mu = mu + Dql(n+1:end);
156  if max(abs(Dql)) < tol
157  break;
158  end
159  end

161  end
