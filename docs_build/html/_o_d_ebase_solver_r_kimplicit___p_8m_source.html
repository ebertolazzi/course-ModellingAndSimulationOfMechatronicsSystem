36  % for the implicit RK methods
37  %
38  % x(k+1) = x(k) + sum(j) b(j) * K(k)
39  %
40  % K(i) = h*f( t(k) + c(i) * dt, x(k) + sum(j) A(i,j) K(j) ), i=1..s
41  %
42  % define the nonlinear system
43  %
44  % / K(1) - h * f( t(k)+c(1)*dt, x(k) + sum(j) A(1,j) K(j) ) \
45  % residual( K ) = | .... |
46  % \ K(s) - h * f( t(k)+c(s)*dt, x(k) + sum(j) A(s,j) K(j) ) /
47  %
48  %
49  function R = stepResidual( self, K )
50  ns = length(self.c);
51  nx = length(self.x0);
52  R = zeros(ns*nx,1);
53  idx = 1:nx;
54  for i=1:ns
55  tmp = self.x0;
56  jdx = 1:nx;
57  for j=1:ns
58  tmp = tmp + self.A(i,j) * K(jdx);
59  jdx = jdx + nx;
60  end
61  ti = self.t0 + self.c(i) * self.dt;
62  R(idx) = K(idx) - self.dt * self.odeClass.f( ti, tmp );
63  idx = idx + nx;
64  end
65  end
66  % - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
67  % Jacobian o the residual
68  %
69  % D residual( K ) / DK
70  %
71  function JR = stepResidualJacobian( self, K )
72  ns = length(self.c);
73  nx = length(self.x0);
74  JR = eye(ns*nx);
75  idx = 1:nx;
76  for i=1:ns
77  tmp = self.x0;
78  jdx = 1:nx;
79  for j=1:ns
80  tmp = tmp + self.A(i,j) * K(jdx);
81  jdx = jdx + nx;
82  end
83  ti = self.t0 + self.c(i) * self.dt;
84  jdx = 1:nx;
85  for j=1:ns
86  JR(idx,jdx) = JR(idx,jdx) - self.dt * self.odeClass.DfDx( ti, tmp );
87  jdx = jdx + nx;
88  end
89  idx = idx + nx;
90  end
91  end

94  function K = solveStepByNewton( self )
95  ns = length( self.c );
96  K0 = self.dt * self.odeClass.f( self.t0, self.x0 );
97  K = repmat( K0(:), ns, 1);
98  fun = @(K) self.stepResidual( K );

100  [K,ierr] = NewtonSolver( fun, jac, K );

104  end
