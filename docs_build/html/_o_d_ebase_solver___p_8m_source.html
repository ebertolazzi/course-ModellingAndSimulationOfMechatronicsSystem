43  % - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
44  %
45  % Solve the problem
46  %
47  % minimize 1/2 * (x-xtilde)^2
48  % subject to h( t, x ) = 0
49  %
50  % Given the Lagrangian
51  %
52  % L( x, lambda ) = 1/2 * (x-xtilde)^2 + lambda * h( t, x )
53  %
54  % solve the problem
55  %
56  % x + DhDx( t, x )^T lambda = xtilde
57  % h( t, x ) = 0
58  %
59  % using the approximation
60  % { h( t_k, x_k ) + DhDx( t_k, x_k ) Dx + O( dx^2 ) = 0 }
61  % define the iterative method
62  %
63  % x_{k+1} = x_k + dx
64  %
65  % where dx is the solution of the linear system
66  %
67  % / I DhDx( t_k, x_k )^T \ / dx \ / xtilde - x_k \
68  % | | | | = | |
69  % \ DhDx( t_k, x_k ) 0 / \ lambda / \ -h( t_k, x_k) /
70  %
71  function x = project( self, t, xtilde )
72  [neq,ninv] = self.odeClass.getDim();
73  x = xtilde;
74  if ninv > 0
75  tol = max(1,norm(xtilde,inf))*sqrt(eps);
76  for k=1:10
77  h = self.odeClass.h( t, x );
78  M = self.odeClass.DhDx( t, x );
79  Dxl = [ eye(neq), M.'; M, zeros(ninv,ninv) ] \ [ xtilde - x;-h];
80  dx = Dxl(1:neq);
81  x = x + dx;
82  if (max(abs(dx)) < tol) && (max(abs(h)) < tol)
83  break;
84  end
85  end
86  end
87  end

89  %
90  % compute approximate solution on a series of point given by the vector t
91  % starting at initial point x0
92  %
93  function x = advance( self, t, x0 )
94  x = zeros(length(x0),length(t));
95  x(:,1) = x0(:);
96  for k=1:length(t)-1
97  % advance
98  xtilde = self.step( t(k), x(:,k), t(k+1)-t(k) );
99  % project solution to the invariants
100  x(:,k+1) = self.project( t(k+1), xtilde );
101  end
102  end
