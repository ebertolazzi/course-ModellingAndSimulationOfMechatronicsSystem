35  % compute approximate solution on a series of point given by the vector t
36  % starting at initial point [p0,v0]
37  %
38  function [pos,vel,lambda] = advance_simple( self, t, p0, v0 )
39  [n,m] = self.DAEclass.getDim();
40  pos = zeros(n,length(t));
41  vel = zeros(n,length(t));
42  lambda = zeros(m,length(t));
43  pos(:,1) = p0(:);
44  vel(:,1) = v0(:);
45  for k=1:length(t)-1
46  tt = t(:,k);
47  pp = pos(:,k);
48  vv = vel(:,k);
49  % get M, Phi, Phi_q, f and b
50  M = self.DAEclass.M( tt, pp );
51  Phi_q = self.DAEclass.Phi_q( tt, pp );
52  f = self.DAEclass.gforce( tt, pp, vv );
53  b = self.DAEclass.b( tt, pp, vv );
54  % solve for v' and lambda
55  vl = [ M, Phi_q.'; ...
56  Phi_q, zeros(m,m) ]\[f;b];
57  vp = vl(1:n);
58  % update using Taylor
59  DT = t(k+1)-t(k);
60  pos(:,k+1) = pp + DT*vv + (DT^2/2)*vp;
61  vel(:,k+1) = vv + DT*vp;
62  lambda(:,k+1) = vl(n+1:n+m);
63  end
64  end

88  f = self.DAEclass.gforce( tt, pp, vv );
89  b = self.DAEclass.b( tt, pp, vv ) + t1 * ( Phi_t + Phi_q * vv ) + t2 * Phi;
90  % solve for v' and lambda
